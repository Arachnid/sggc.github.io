contract IndexOf {
    event Move(uint);
    function indexOf(string haystack, string needle) public pure returns(int) {
        uint n = bytes(needle).length - 1;
        uint h = bytes(haystack).length - 1;
        if (bytes(needle).length == 0) {
            return 0;
        }
        if (bytes(haystack).length == 0) {
            return -1;
        }
        if (n > h) {
            return -1;
        }
        
        uint indexH = n;
        while (true) {
            if (indexH > h) {
                return -1;
            }
            uint indexN = n;
            while (indexN >= 0) {
                if (bytes(needle)[indexN] == bytes(haystack)[indexH]) {
                    if (indexN == 0) {
                        return int(indexH);
                    } else {
                        indexN--;
                        indexH--;
                        continue;
                    }
                } else {
                    break;
                }
            }

            indexH = indexH + n - indexN + 1;
        }

        return -1;
    }
}
