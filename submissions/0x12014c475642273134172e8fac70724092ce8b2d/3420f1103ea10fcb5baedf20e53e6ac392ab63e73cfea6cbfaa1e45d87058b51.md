pragma solidity 0.4.24;

contract IndexOf {
    function indexOf(string haystack, string needle) public pure returns(int) {
        uint[74] memory F;
        uint[2] memory g;

        if (bytes(needle).length == 0) {
            return 0;
        }

        uint n = bytes(haystack).length;
        uint m = bytes(needle).length;

        if (n < m) {
            return -1;
        }
        F[0] = 0;
        F[1] = 0;

        for (g[0] = 1; g[0] < m; ) {
            g[1] = F[g[0]];
            while (true) {
                if (bytes(needle)[g[1]] == bytes(needle)[g[0]]) {
                    ++g[0];
                    F[g[0]] = g[1] + 1;
                    break;
                }
                if (g[1] == 0) {
                    ++g[0];
                    F[g[0]] = 0;
                    break;
                }
                ++g[0];
                g[1] = F[g[1]];
            }
        }
        
        g[0] = 0;
        g[1] = 0;

        while (g[0] < n) {
            while (bytes(haystack)[g[0]] == bytes(needle)[g[1]]) {
                ++g[0];
                ++g[1];

                if (g[1] == m) {
                    return int(g[0] - m);
                }
                if (g[0] >= n) {
                    break;
                }
                continue;
            }
            if (g[1] > 0) {
                g[1] = F[g[1]];
                continue;
            }
            ++g[0];
        }

        return -1;
    }
}
